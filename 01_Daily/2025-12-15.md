## 📅 이번 주 남은 일정
```dataviewjs
const gcal = app.plugins.plugins["google-calendar"];

if (!gcal || !gcal.api) {
    dv.paragraph("❌ Google Calendar를 불러올 수 없습니다.");
} else {
    const today = moment().startOf('day');
    const endOfWeek = moment().endOf('week').day(7);
    
    let output = `🔍 **디버깅 정보**\n`;
    output += `- 오늘: ${moment().format('YYYY-MM-DD HH:mm')}\n`;
    output += `- 검색 시작: ${today.format('YYYY-MM-DD HH:mm')}\n`;
    output += `- 검색 종료: ${endOfWeek.format('YYYY-MM-DD HH:mm')}\n\n`;
    
    try {
        const events = await gcal.api.getEvents(today.toDate(), endOfWeek.toDate());
        
        output += `📊 조회 결과: ${events ? events.length : 0}개\n\n`;
        
        // 모든 이벤트 원본 데이터 출력
        if (events && events.length > 0) {
            output += `**원본 데이터:**\n`;
            events.forEach((e, idx) => {
                output += `\n[${idx + 1}] ${e.summary}\n`;
                output += `  - start: ${JSON.stringify(e.start)}\n`;
                output += `  - end: ${JSON.stringify(e.end)}\n`;
            });
            output += `\n---\n\n`;
            
            // 정렬
            events.sort((a, b) => {
                const aTime = a.start?.dateTime || a.start?.date;
                const bTime = b.start?.dateTime || b.start?.date;
                return new Date(aTime) - new Date(bTime);
            });
            
            let currentDate = "";
            
            for (const e of events) {
                const eventStart = e.start?.dateTime || e.start?.date;
                const eventDate = moment(eventStart).format("MM/DD (ddd)");
                
                if (eventDate !== currentDate) {
                    if (currentDate !== "") output += "\n";
                    output += `**${eventDate}**\n`;
                    currentDate = eventDate;
                }
                
                if (e.start?.date) {
                    output += `- 🗓️ ${e.summary}\n`;
                } else if (e.start?.dateTime) {
                    const start = moment(e.start.dateTime).format("HH:mm");
                    const end = moment(e.end.dateTime).format("HH:mm");
                    output += `- ⏰ ${start} ~ ${end} ${e.summary}\n`;
                }
            }
        } else {
            output += "🎉 이번 주 남은 일정이 없습니다.\n\n";
            output += "**확인사항:**\n";
            output += "- Google Calendar 웹에서 일정이 보이나요?\n";
            output += "- Obsidian의 Google Calendar 플러그인 설정에서 올바른 캘린더가 선택되었나요?\n";
        }
    } catch (error) {
        output += `❌ 오류 발생: ${error.message}\n`;
        output += `Stack: ${error.stack}\n`;
    }
    
    dv.paragraph(output);
}
```
